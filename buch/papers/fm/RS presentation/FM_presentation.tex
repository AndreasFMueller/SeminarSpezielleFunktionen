\section{Proof}
\begin{frame}
	\frametitle{Bessel}

	\visible<1->{\begin{align}
			\cos(\beta\sin\varphi)
			&=
			J_0(\beta) + 2\sum_{m=1}^\infty J_{2m}(\beta) \cos(2m\varphi)
			\\
			\sin(\beta\sin\varphi)
			&=
			J_0(\beta) + 2\sum_{m=1}^\infty J_{2m}(\beta) \cos(2m\varphi)
			\\
			J_{-n}(\beta) &= (-1)^n J_n(\beta)
		\end{align}}
	\visible<2->{\begin{align}
			\cos(A + B) 
			&= 
			\cos(A)\cos(B)-\sin(A)\sin(B)
			\\
			2\cos (A)\cos (B)
			&=
			\cos(A-B)+\cos(A+B)
			\\
			2\sin(A)\sin(B)
			&=
			\cos(A-B)-\cos(A+B)
		\end{align}}
\end{frame}
