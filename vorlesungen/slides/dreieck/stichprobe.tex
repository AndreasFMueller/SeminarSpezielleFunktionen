\begin{columns}[t,onlytextwidth]
\begin{column}{0.48\textwidth}
\begin{block}{Zufallsvariable}
Gegeben eine Zufallsvariable $X$ mit
Verteilungsfunktion
\[
F_X(x)
=
P(X\le x)
\]
und
Wahrscheinlichkeitsdichte
\[
\varphi_X(x)
=
\frac{d}{dx} F_X(x)
\]
\end{block}
\begin{block}{Gleichverteilung}
\[
F(x) = \begin{cases}
0&\qquad x<0\\
x&\qquad 0\le x \le 1\\
1&\qquad 1<x
\end{cases}
\qquad\Rightarrow\qquad
\varphi(x)
=
\begin{cases}
1&\qquad 0\le x \le 1\\
0&\qquad\text{sonst}.
\end{cases}
\]
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Stichprobe}
$n$ Zufallsvariablen $X_1,\dots,X_n$
\begin{itemize}
\item
alle $X_i$ haben die gleiche Verteilung wie $X$
\item
die $X_i$ sind unabhängig
\end{itemize}
\end{block}
\end{column}
