\begin{block}{Taylor-Reihe}
\begin{align*}
\sin t^2
&=
\sum_{k=0}^\infty
(-1)^k \frac{t^{2k+1}}{(2k+1)!}
\\
%\int \sin t^2\,dt
\uncover<2->{
S(t)
&=
\sum_{k=0}^\infty
(-1)^k \frac{t^{4k+3}}{(2k+1)!(4n+3)}
}
\\
\cos t^2
&=
\sum_{k=0}^\infty
(-1)^k \frac{t^{2k}}{(2k)!}
\\
%\int \sin t^2\,dt
\uncover<3->{
C(t)
&=
\sum_{k=0}^\infty
(-1)^k \frac{t^{4k+1}}{(2k)!(4k+1)}
}
\end{align*}
\end{block}
